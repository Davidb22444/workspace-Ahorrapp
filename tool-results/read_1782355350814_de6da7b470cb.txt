   133→// ── Enhanced Tooltip ──────────────────────────────────────────────
   134→function EnhancedChartTooltip({ active, payload, label }: any) {
   135→  if (active && payload && payload.length) {
   136→    const total = payload.reduce((sum: number, p: any) => sum + (p.value || 0), 0)
   137→    return (
   138→      <div className="bg-card border border-border rounded-xl p-3 shadow-xl">
   139→        <p className="text-sm font-semibold text-foreground mb-1.5">{label}</p>
   140→        {payload.map((entry: any, idx: number) => (
   141→          <div key={idx} className="flex items-center justify-between gap-3 text-xs">
   142→            <div className="flex items-center gap-1.5">
   143→              <div className="w-2 h-2 rounded-full" style={{ background: entry.color }} />
   144→              <span className="text-muted-foreground">{entry.name}</span>
   145→            </div>
   146→            <span className="font-semibold tabular-nums">{formatCurrency(entry.value)}</span>
   147→          </div>
   148→        ))}
   149→        {payload.length > 1 && (
   150→          <div className="border-t border-border/50 mt-1.5 pt-1.5 flex items-center justify-between text-xs">
   151→            <span className="text-muted-foreground font-medium">Net</span>
   152→            <span className={cn('font-bold tabular-nums', total >= 0 ? 'text-emerald-500' : 'text-rose-500')}>
   153→              {formatCurrency(total)}
   154→            </span>
   155→          </div>
   156→        )}
   157→      </div>
   158→    )
   159→  }
   160→  return null
   161→}

   163→// ── Emoji mapper for savings goals ─────────────────────────────────
   164→function getGoalEmoji(name: string): string {
   165→  const lower = name.toLowerCase()
   166→  if (lower.includes('emergency') || lower.includes('emergencia')) return '🛡️'
   167→  if (lower.includes('vacat') || lower.includes('viaje')) return '✈️'
   168→  if (lower.includes('laptop') || lower.includes('computer') || lower.includes('pc')) return '💻'
   169→  if (lower.includes('car') || lower.includes('auto') || lower.includes('coche')) return '🚗'
   170→  if (lower.includes('house') || lower.includes('home') || lower.includes('down') || lower.includes('casa')) return '🏠'
   171→  if (lower.includes('wedding') || lower.includes('boda')) return '💍'
   172→  if (lower.includes('education') || lower.includes('study') || lower.includes('college')) return '🎓'
   173→  if (lower.includes('retire') || lower.includes('jubil')) return '🌴'
   174→  if (lower.includes('gift') || lower.includes('regalo')) return '🎁'
   175→  if (lower.includes('invest') || lower.includes('inversion')) return '📈'
   176→  return '🎯'
   177→}

   438→  const mapApiData = (api: any): DashboardData => ({

   448→    expenseCategories: (api.categoryBreakdown ?? []).map((c: any) => ({

   450→      value: c.total ?? c.value ?? 0,
   451→      color: c.color || '#94a3b8',
   452→    })),
   453→    budgetVsActual: [] as DashboardData['budgetVsActual'],
   454→    recentTransactions: (api.recentMovements ?? []).map((m: any) => ({
   455→      id: m.id,
   456→      type: m.type || 'expense',
   457→      description: m.description,
   458→      amount: m.type === 'entrada' ? m.amount : -m.amount,
   459→      date: m.date,
   460→      category: m.category?.name,
   461→      categoryColor: m.category?.color,
   462→    })),
   463→    savingsGoals: (api.savingsProgress ?? []).map((g: any) => ({
   464→      id: g.id,
   465→      name: g.name,
   466→      saved: g.savedAmount ?? g.saved ?? 0,
   467→      target: g.targetAmount ?? g.target ?? 0,
   468→      deadline: g.deadline,
   469→    })),
   470→  })
   471→
   472→  const doFetchData = async () => {
   473→    setLoading(true)
   474→    try {
   475→      if (user?.id) {
   476→        const currentMonth = format(new Date(), 'yyyy-MM')
   477→        const [dashRes, budgetRes, dailyRes] = await Promise.all([
   478→          fetch(`/api/dashboard?accountId=${user.id}`),
   479→          fetch(`/api/budget?accountId=${user.id}`),
   480→          fetch(`/api/dashboard/daily-spending?accountId=${user.id}&month=${currentMonth}`),
   481→        ])
   482→        if (dailyRes.ok) {
   483→          const dailyJson = await dailyRes.json()
   484→          setDailySpendingData(dailyJson.days ?? [])
   485→        }
   486→        if (dashRes.ok) {
   487→          const json = await dashRes.json()
   488→          const mapped = mapApiData(json)
   489→
   490→          if (budgetRes.ok) {
   491→            try {
   492→              const budgetJson = await budgetRes.json()
   493→              const budgets = budgetJson.budgets || []
   494→              if (budgets.length > 0) {
   495→                const b = budgets[0]
   496→                const totalIncome = mapped.totalIncome || b.totalAmount || 5200
   497→                const needsPct = b.needsPercent ?? 50
   498→                const wantsPct = b.wantsPercent ?? 30
   499→                const savingsPct = b.savingsPercent ?? 20
   500→                mapped.budgetVsActual = [
   501→                  { name: 'Needs', planned: Math.round(totalIncome * needsPct / 100), actual: Math.round(mapped.totalExpenses * needsPct / 100) },
